    \begin{longtable}[]{@{}ll@{}}
\toprule()
Plot Type & NaN Handling \\
\midrule()
\endhead
Line & Leave gaps at NaNs \\
Line & (stacked) Fill 0's \\
Bar & Fill 0's \\
Scatter & Drop NaNs \\
Histogram & Drop NaNs (column-wise) \\
Box & Drop NaNs (column-wise) \\
Area & Fill 0's \\
KDE & Drop NaNs (column-wise) \\
Hexbin & Drop NaNs \\
Pie & Fill 0's \\
\bottomrule()
